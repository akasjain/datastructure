function cost = squaredErrorCost(A, b, x)
  % Your code here
  first = A * x;
  num = first - b;
  cost = norm(num).^2;
endfunction

function A_inv_b = matrixInverseVector(A, b, x_init, alpha)
  % Your code here
  
  while true,
    ax = A * x_init;
    second_term = ax - b;
    product = A * second_term
    scalar = 2*alpha;
    if norm(second_term) < 0.001,
      A_inv_b = x_init;
      break
    end
    
    x_init = x_init - product.*scalar
  end
  
endfunction
